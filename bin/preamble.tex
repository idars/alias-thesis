% Custom titlepage
\newcommand*{\maketitlepage}[4]{
    \thispagestyle{empty}
    
    \begingroup
        \fontfamily{phv} \selectfont
        \centering % Center all text
        
        \vspace{200pt}
        
        {\Huge #1} \\ % Title
        
        \vspace{5pt}
        
        {\Large \textsl{#2}} \\ % Subtitle or further description
        
        \vspace{50pt}
        
        {\Large #3} \\ % Author name
        
        \vfill % Whitespace between the author name and the publisher logo
        
        {\Large Master's thesis in Software Engineering at \\
        
        \vspace{10pt}
        
        Department of Computing, Mathematics and Physics, \\
        Western Norway University of Applied Sciences \\
        
        \vspace{10pt}
        
        Department of Informatics, \\
        University of Bergen \\}
        
        \vspace{10pt}
        
        {\large #4} % Month and year published
        
        \vspace{10pt} % to compensate for the lack of captions
        
        \begin{figure}[h!]
            \begin{subfigure}{200pt}
                \includegraphics[width=\linewidth, valign=c]{hvl-logo.png}
            \end{subfigure}
            \hfill
            \begin{subfigure}{70pt}
                \includegraphics[width=\linewidth, valign=c]{uib-logo.eps}
            \end{subfigure}
        \end{figure}
        
        \vspace{10pt} % to compensate for the lack of captions
    \endgroup
    
    \pagebreak
}
